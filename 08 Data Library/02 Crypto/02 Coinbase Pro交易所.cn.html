Coinbase Pro交易所是一个总部在美国的加密货币交易所，以前称为全球数字资产交易所（GDAX）。今天，Coinbase Pro提供9个货币对，涵盖BTC、EUR、LTC、USD和ETH的组合。QuantConnect Coinbase Pro数据的时间戳精度为毫秒。

必须使用AddCrypto()方法手动地在Initialize()方法中添加数据。 

// 初始化所需的数据和分辨率，以及算法的现金和开始结束日期。所有算法都必须初始化。
public override void Initialize()
{
    SetStartDate(2015, 10, 07);  //设置开始日期
    SetEndDate(2017, 10, 11);    //设置结束日期
    SetCash(100000);             //设置策略的金额
    AddCrypto("BTCUSD", Resolution.Hour, Market.GDAX);
}

// 访问请求的数据
public override void OnData(Slice data) {
    // 通过tradebar字典(symbol - bar)
    data.Bars["BTCUSD"].Close
    // 或通过一个ticks列表:
    data.Ticks["BTCUSD"][0].Close
}

# 初始化所需的数据和分辨率，以及算法的现金和开始结束日期。所有算法都必须初始化。
def Initialize(self):
    self.SetStartDate(2015, 10, 07)  # 设置开始日期
    self.SetEndDate(2017, 10, 11)    # 设置结束日期
    self.SetCash(100000)             # 设置策略的金额
    self.AddCrypto("BTCUSD", Resolution.Hour, Market.GDAX)

# 访问请求的数据
def OnData(self, data):
    # 通过tradebar字典(symbol - bar)
    data.Bars["BTCUSD"].Close
    # 或通过一个ticks列表:
    data.Ticks["BTCUSD"][0].Close
